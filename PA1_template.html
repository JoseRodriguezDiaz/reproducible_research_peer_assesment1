imputdata <- numeric()
for (i in 1:nrow(activity)) {
  obs <- activity[i, ]
  if (is.na(obs$steps)) {
    steps <- subset(newsteps, interval == obs$interval)$steps
  } else {
    steps <- obs$steps
  }
  imputdata <- c(imputdata, steps)
}
